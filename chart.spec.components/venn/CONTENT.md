 - padding --> 绘图区域的边距，默认15
 - nodeField --> 节点字段配置
 - valueField --> 值字段配置
 - labelField --> 文案字段配置
 - text 
   + enabled --> 是否出现文案
   + fillStyle --> 文案颜色
   + fontSize --> 文案大小
   + textAlign --> 文案左右对齐配置
   + textBaseline --> 文案垂直对齐配置
   + format --> format格式函数
 - circle 
   + fillAlpha --> 圆透明度，默认0.25
   + fillStyle --> 圆填充色
   + strokeStyle --> 圆描边色
   + lineWidth --> 圆边框大小，默认2
